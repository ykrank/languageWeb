1 概述

加密算法可以分为单向加密，对称加密，和非对称加密几种。

单向加密算法

单向加密算法是非可逆加密，就是不可解密的加密方法。MD5、SHA属于单项加密，可以使用国密SM3替换。Hmac是比较复杂的单向加密，产品提供了Hmac-MD5和Hmac-SHA1两种加密模式,均可以使用Hmac-SM3加密模式来替换。

对称加密算法

对称加密算法是可逆加密，在加密和解密时使用的是同一个秘钥。DES和AES加密属于对称加密，可以使用国密SM4替换。

非对称加密 > 非对称加密是可逆加密，需要两个密钥来进行加密和解密，这两个秘钥是公开密钥和私有密钥。RSA加密属于非对称加密，可以使用国密SM2替换。

    /* SM3加密。 */
    Array<uint8, 1>^ ERTCrypto::SM3Encrypt(const Array<uint8, 1>^ data)
    {
        int length = data->Length;
        unsigned char *c = ArrayCXToCPlus(data);

        ERTCryptoSM3 sm3;
        ERTUserData out = sm3.digestFromData(c, length);
        unsigned char *cr = out.getData();
        int outlength = out.count();

        Array<uint8, 1>^ result = ArrayCPlusToCX(cr, outlength);

        delete[]c;
        return result;
    }

    /* Hmac-SM3加密。 */
    Array<uint8, 1>^ ERTCrypto::SM3EncrypToHmac(const Array<uint8, 1>^ inputBytes, const Array<uint8, 1>^ keyBytes)
    {
        unsigned char *input = ArrayCXToCPlus(inputBytes);
        const char *key = ArrUint8ToConstChar(keyBytes);

        ERTCryptoHMAC hmac;
        hmac.setEvpEngine(ERTCryptoHMAC::EVP_MD_sm3);
        hmac.setKey(key, keyBytes->Length);

        ERTUserData out = hmac.digestFromData(input, inputBytes->Length);

        unsigned char *resultChar = out.getData();
        int resultLength = out.count();
        Array<uint8, 1>^ result = ArrayCPlusToCX(resultChar, resultLength);

        return result;
    }

    /* 验证证书的有效性 */ //RSA
    int ERTCrypto::SM2VerifyCertificate(const Array<uint8, 1>^ vertifyBytes, const Array<uint8, 1>^ rootBytes)
    {
        unsigned char *vertifyUC = ArrayCXToCPlus(vertifyBytes);
        unsigned char *rootUC = ArrayCXToCPlus(rootBytes);

        ERTUserData rootData, verifyData;
        rootData.append(rootUC, rootBytes->Length);
        verifyData.append(vertifyUC, vertifyBytes->Length);

        X509 *xRoot509 = ERTCryptoX509::derDataToX509(rootData);
        X509 *xVertify509 = ERTCryptoX509::derDataToX509(verifyData);
        bool isRoot = ERTCryptoX509::addCert(xRoot509);
        if (!isRoot) return -1;

        bool isVertify = ERTCryptoX509::verify(xVertify509);
        if (!isVertify) return -1;

        ERTCryptoX509::cleanX509(xVertify509);
        return isVertify;
    }

    int ERTCrypto::Ert509CaVerify(const Array<uint8, 1>^ rootData, const Array<uint8, 1>^ verifyData)
    {
        unsigned char* rootChar = ArrayCXToCPlus(rootData);
        unsigned char* verifyChar = ArrayCXToCPlus(verifyData);

        ERTUserData rootCa;
        rootCa.append(rootChar, rootData->Length);
        ERTUserData verifyCa;
        verifyCa.append(verifyChar, verifyData->Length);

        ERTX509RootCa ca;
        if (!ca.readCaCer(rootCa))
        {
            return -1;
        }
        bool res = ca.verifyX509(verifyCa);
        return res;
    }
