2 生成菜单数据

根据界面所需的json数据我们需要在helloapp_channel的list_collection_items方法中实现返回的json数据。

在此方法中具体需要关注的为：CsKey, Input。

CsKey cs模板名称，在此时获取collection/list中为helloapp_channel_i。其中helloapp为当前创建app的名称，i为客户端类型，i表示为高级客户端，比如android，iphone，winphone等。

Input 传入cs模板的tuplist参数。一般为此种格式：

[
{channel,
[{id,"helloapp_inner_tranfer"},
{name,[232,161,140,229,134,133,232,189,172,232,180,166]},
{url,"http://192.168.241.3:4002/channel/run?id=helloapp_inner_tranfer"},
{secure,0}]},
{channel,[
{id,"helloapp_other_tranfer"},
{name,[232,183,168,232,161,140,232,189,172,232,180,166]},
{url,"http://192.168.241.3:4002/channel/run?id=helloapp_other_tranfer"},
{secure,0}]}
]

创建helloapp_channels_i.cs文件

#{cs set:count = 0}#
#{cs each:c = channel}#
#{cs     set:count = count+1}#
#{cs /each}#
{
"return":
{
"error_code":"000000",
"error_msg":"",
"channel_list":
[
#{cs set:index = 0}#
#{cs each:c=channel}#
#{cs     set:index = index+1}#
{
"channel":"#{cs var:c.id}#",
"name":"#{cs var:c.name}#"
#{cs if:index == count}#
}
#{cs else}#
},
#{cs /if}#            
#{cs /each}#
]
}
}

    --二级菜单界面请求
    local function channel_callback(response)
        local responseCode = response["responseCode"];
        local collection_id = response["collection_id"];
        if responseCode == 200 then
            local json_data = response["responseBody"];
            global_tab[collection_id] = json_data;
            --异步请求账户列表界面跳转入账户列表界面
            http:postAsyn(nil,"test_s/get_page","name=training/channel_entrance.xhtml",qry_page_callback,nil);
        else
            window:alert("请求服务器出错！");
        end;
    end;

    --请求二级菜单
    function post_channel(collection_id)
        --异步请求账户查询获取账户列表存入global_tab
        global_tab["collection_id"] = collection_id;

        http:postAsyn(nil,"app_s/list","id="..collection_id.."&type=channels",channel_callback, {collection_id = collection_id});
    end;
