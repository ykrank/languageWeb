1.1 1.o/1

(1)描述： 根据指定键获取指定数据结构中对应的值。数据结构可为EWP Channel、Collection和Proplist

o(X, _Field) when X == undefined orelse X == []->
    undefined;

o({mnesia, channels, Record}, Field) ->
    case ewp_mnesia:get(channels, Field, Record) of
        undefined ->
            proplists:get_value(Field, ewp_mnesia:get(channels, props, Record));
        V ->
            V
    end;

o({mnesia, Tab, Record}, Field) ->
    ewp_mnesia:get(Tab, Field, Record);


o({_Type, _Tab, PropList}, Field) ->
    proplists:get_value(Field, PropList);

o(PropList, Field) when is_list(PropList)->
    proplists:get_value(Field, PropList).

(2)参考数据类型：

o(proplist()|channel(), Field::atom()) -> any()
proplist() = [{Key :: term(), Value :: term()}]

(3)编译运行效果为：

创建record:
(ebank@tracvm)1> rd(user,{id,name}).                 
user

(ebank@tracvm)2> Person=#user{}.    
#user{id = undefined,name = undefined}

创建表：
(ebank@tracvm)3> mnesia:create_table(user, [{disc_only_copies, nodes()}, {attributes, record_info(fields,user)}]).    
{atomic,ok}

效果为：
(ebank@tracvm)4> evo:o({mnesia,user,Person#user{id=1,name="Lulu"}},name).
"Lulu"

或者为：
(ebank@tracvm)5> evo:o([{id,123}],id).
123

1.2 2.o/3

(1)描述：根据指定键获取指定数据结构中对应的值，无法获取时返回指定的默认值。

o(X, Field, Default) ->
    case o(X, Field) of
        undefined ->
            Default;
        V ->
            V
    end.

(2)参考数据类型：

o(proplist()|channel(), Field::atom(), Default :: term()) -> any()

(3)编译运行效果为：

a.正确获取时：
(ebank@tracvm)1> evo:o([{id,123}],id,false).
123

b.无法获取时：
(ebank@tracvm)2> evo:o([{id,123}],idsss,false).
false

1.3 3.ot/2

(1)描述：根据指定键获取指定数据结构中对应的值，无法获取时会抛出错误。

ot(Proplist, Key) ->
    case o(Proplist, Key) of
    undefined ->
        throw({error, lists:concat(["param ", Key, " not found in ot/2"])});
    Val ->
        Val
    end.

(2)参考数据类型：

(proplist()|channel(), Field::atom()) -> any()
抛出错误时： {error, string()}

(3)编译运行效果为：

a.正确获取时：
(ebank@tracvm)44> evo:ot([{id,123}],id).
123

b.无法获取时：
(ebank@tracvm)3> evo:ot([{id,123}],idsss).
** exception throw: {error,"param idsss not found in ot/2"}
